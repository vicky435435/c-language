2. What will be the output of the C program?

#include<stdio.h>

{
	void  num=10;

3. What will be the output of the C program?

#include<stdio.h>
int main()
{
	printf("%d\t",sizeof(2.5));
	printf("%d\t",sizeof(2));
	printf("%d",sizeof('A'));
	return 0;
}
